    ```python
    def get_value_by_path(data: dict, path: str, default=None):
        """Извлекает значение по 'a.b.c' — копия из eval_rag_metrics.py:338-352."""
        if not path:
            return default
        keys = path.split('.')
        val = data
        for k in keys:
            if isinstance(val, dict):
                val = val.get(k)
            elif isinstance(val, list) and k.isdigit():
                val = val[int(k)]
            else:
                return default
            if val is None:
                return default
        return val

    def resolve_template(template, rec: dict):
        """Рекурсивная подстановка {{var}} — копия из eval_rag_metrics.py:354-368."""
        if isinstance(template, str):
            res = template
            for k, v in rec.items():
                if isinstance(v, str):
                    res = res.replace(f"{{{{{k}}}}}", v)
            return res
        elif isinstance(template, dict):
            return {k: resolve_template(v, rec) for k, v in template.items()}
        elif isinstance(template, list):
            return [resolve_template(v, rec) for v in template]
        return template
    ```

    ```python
                cls = getattr(mod, attr)
                if not isinstance(cls, type):
                    continue

                if issubclass(cls, MetricWithLLM) and issubclass(cls, SingleTurnMetric):
    ```

    **7. save_results(records, result, run_dir, metadata) -> Path** — per D-06:
    ```python
    def save_results(records: list[dict], result, run_dir: Path) -> Path:
        """Сохраняет metrics.json в формате совместимом с DeepEval UI (phase 2 читает этот файл)."""
        df = result.to_pandas()
        # RAGAS возвращает колонки: faithfulness, answer_relevancy, context_precision,
        # context_recall, answer_correctness + (кастомные) — все в snake_case
        rows = []
        scores_list = df.to_dict(orient="records")
        for rec, scores in zip(records, scores_list):
            row = {
                "session_id":       rec.get("session_id") or rec.get("id"),
                "category":         rec.get("category"),
                "intent":           rec.get("intent"),
                "user_query":       rec.get("user_query"),
                "actual_answer":    rec.get("actual_answer"),
                "expected_answer":  rec.get("expected_answer"),
                "retrieval_context": rec.get("retrieval_context") or [],
            }
            # Маппинг RAGAS-колонок в *_score поля (формат как у DeepEval)
            for ragas_key, out_key in [
                ("faithfulness",       "faithfulness_score"),
                ("answer_relevancy",   "answer_relevancy_score"),
                ("context_precision",  "context_precision_score"),
                ("context_recall",     "context_recall_score"),
                ("answer_correctness", "answer_correctness_score"),
            ]:
                val = scores.get(ragas_key)
                # NaN -> None (json.dump совместимость + RAGAS-04 gracefully skip)
                if val is not None and (val != val):  # NaN check
                    val = None
                row[out_key] = val
            # Кастомные метрики: все остальные колонки (не в стандартных)
            standard = {"user_input", "response", "retrieved_contexts", "reference",
                        "faithfulness", "answer_relevancy", "context_precision",
                        "context_recall", "answer_correctness"}
            for k, v in scores.items():
                if k in standard:
                    continue
                if v is not None and (v != v):
                    v = None
                row[f"{k}_score"] = v
            rows.append(row)
        out_path = run_dir / "metrics.json"
        with open(out_path, "w", encoding="utf-8") as f:
            json.dump(rows, f, ensure_ascii=False, indent=2)
        return out_path
    ```

    ```python
        print(f"metrics.json  → {out_path}")
    ```
